// db/travelDB.ts
import Dexie, { Table } from 'dexie';

export interface TravelPost {
  id?: number;
  title: string;
  content: string;
  images: string[];
  location: {
    lat: number;
    lng: number;
    name: string;
  };
  createdAt: Date;
  syncStatus: 'synced' | 'pending' | 'failed';
}

export class TravelDB extends Dexie {
  posts!: Table;
  images!: Table<{ id?: number; postId: number; blob: Blob; url: string }>;

  constructor() {
    super('TravelDB');
    this.version(1).stores({
      posts: '++id, title, createdAt, syncStatus',
      images: '++id, postId, url'
    });
  }
}

export const db = new TravelDB();

// Service for offline operations
export class OfflineTravelService {
  async saveTravelPost(post: Omit) {
    const id = await db.posts.add({
      ...post,
      syncStatus: 'pending'
    });
    
    // Schedule background sync
    if ('serviceWorker' in navigator && 'sync' in window.ServiceWorkerRegistration.prototype) {
      const registration = await navigator.serviceWorker.ready;
      await registration.sync.register('sync-travel-posts');
    }
    
    return id;
  }

  async getTravelPosts(limit = 20, offset = 0) {
    return await db.posts
      .orderBy('createdAt')
      .reverse()
      .offset(offset)
      .limit(limit)
      .toArray();
  }

  async searchPosts(query: string) {
    return await db.posts
      .filter(post => 
        post.title.toLowerCase().includes(query.toLowerCase()) ||
        post.content.toLowerCase().includes(query.toLowerCase())
      )
      .toArray();
  }
}

RESTful API Design Principles

Resource-Based URLs

Use nouns for resources, not verbs for actions.


// Good: Resource-based
GET    /api/travel-posts
POST   /api/travel-posts
GET    /api/travel-posts/:id
PUT    /api/travel-posts/:id
DELETE /api/travel-posts/:id

// Bad: Action-based
GET    /api/getTravelPosts
POST   /api/createTravelPost
HTTP Status Codes

Use appropriate HTTP status codes for different scenarios.


// Success responses
200 - OK (GET, PUT)
201 - Created (POST)
204 - No Content (DELETE)

// Client error responses
400 - Bad Request
401 - Unauthorized
403 - Forbidden
404 - Not Found
422 - Unprocessable Entity

// Server error responses
500 - Internal Server Error
503 - Service Unavailable
API Versioning Strategy
  • URL Versioning: /api/v1/travel-posts
  • Header Versioning: Accept: application/vnd.api+json;version=1
  • Semantic Versioning: Follow semver for API versions
  • Deprecation Policy: Support previous version for 6 months

import { ApifyApi } from 'apify-client';
import { Actor } from 'apify';

class TravelContentScraper {
  constructor() {
    this.apifyClient = new ApifyApi({
      token: process.env.APIFY_API_TOKEN,
    });
  }

  async scrapeInstagramTravelPosts(hashtags, maxPosts = 100) {
    try {
      const input = {
        hashtags: hashtags,
        resultsLimit: maxPosts,
        addParentData: false,
        enhanceUserInformation: true,
        isUserTaggedFeedURL: false,
        onlyPostsWithLocation: true, // Focus on travel posts with location
        scrapePostsUntilDate: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000).toISOString()
      };

      const run = await this.apifyClient.actor('apify/instagram-hashtag-scraper').call(input);
      const { items } = await this.apifyClient.dataset(run.defaultDatasetId).listItems();
      
      return this.processInstagramData(items);
    } catch (error) {
      console.error('Instagram scraping failed:', error);
      throw new Error('Failed to scrape Instagram content');
    }
  }

  async scrapeTravelBlogs(urls) {
    const input = {
      startUrls: urls.map(url => ({ url })),
      maxRequestsPerCrawl: 1000,
      maxConcurrency: 10,
      requestHandlerTimeoutSecs: 60,
      customData: {
        extractImages: true,
        extractLocation: true,
        extractTravelTips: true
      }
    };

    const run = await this.apifyClient.actor('apify/web-scraper').call(input);
    const { items } = await this.apifyClient.dataset(run.defaultDatasetId).listItems();
    
    return this.processBlogData(items);
  }

  processInstagramData(items) {
    return items.map(item => ({
      id: item.id,
      caption: item.caption,
      images: item.displayUrl ? [item.displayUrl] : [],
      location: item.locationName || null,
      coordinates: item.lat && item.lng ? { lat: item.lat, lng: item.lng } : null,
      hashtags: item.hashtags || [],
      timestamp: new Date(item.timestamp),
      engagement: {
        likes: item.likesCount || 0,
        comments: item.commentsCount || 0
      },
      source: 'instagram',
      author: {
        username: item.ownerUsername,
        fullName: item.ownerFullName
      }
    }));
  }

  processBlogData(items) {
    return items.map(item => ({
      title: item.title || this.extractTitleFromContent(item.text),
      content: item.text,
      url: item.url,
      images: item.images || [],
      publishedDate: this.extractDateFromContent(item.text),
      location: this.extractLocationFromContent(item.text),
      source: 'blog',
      domain: new URL(item.url).hostname
    }));
  }

  extractTitleFromContent(text) {
    // Extract title from content using regex or NLP
    const titleMatch = text.match(/^(.{1,100})/);
    return titleMatch ? titleMatch[1].trim() : 'Untitled Travel Post';
  }

  extractLocationFromContent(text) {
    // Use regex or NLP to extract location mentions
    const locationPatterns = [
      /(?:in|at|visiting|traveled to)\s+([A-Z][a-z]+(?:\s+[A-Z][a-z]+)*)/g,
      /([A-Z][a-z]+,\s*[A-Z][a-z]+)/g
    ];
    
    for (const pattern of locationPatterns) {
      const matches = text.match(pattern);
      if (matches) {
        return matches[0].replace(/^(in|at|visiting|traveled to)\s+/i, '');
      }
    }
    return null;
  }
}

Blog Content Processor:

class TravelBlogProcessor {
  constructor(openaiClient) {
    this.openaiClient = openaiClient;
  }

  async processExtractedContent(rawContent) {
    const cleanedContent = await this.cleanContent(rawContent);
    const qualityScore = await this.assessQuality(cleanedContent);
    
    if (qualityScore < 0.7) {
      return null; // Skip low-quality content
    }

    const structuredData = await this.extractStructuredData(cleanedContent);
    return {
      ...structuredData,
      qualityScore,
      processedAt: new Date().toISOString()
    };
  }

  async cleanContent(content) {
    // Remove ads, navigation, and irrelevant content
    const cleaned = content
      .replace(/)<[^<]*)*<\/script>/gi, '')
      .replace(/)<[^<]*)*<\/style>/gi, '')
      .replace(//g, '')
      .replace(/<[^>]+>/g, ' ')
      .replace(/\s+/g, ' ')
      .trim();

    return cleaned;
  }

  async assessQuality(content) {
    try {
      const response = await this.openaiClient.chat.completions.create({
        model: "gpt-3.5-turbo",
        messages: [{
          role: "system",
          content: "You are a content quality assessor. Rate the quality of travel content from 0-1 based on informativeness, readability, and travel value."
        }, {
          role: "user",
          content: `Rate this travel content: ${content.substring(0, 1000)}...`
        }],
        max_tokens: 50
      });

      const rating = parseFloat(response.choices[0].message.content.match(/\d+\.?\d*/)?.[0] || '0');
      return Math.min(Math.max(rating, 0), 1);
    } catch (error) {
      console.error('Quality assessment failed:', error);
      return 0.5; // Default neutral score
    }
  }

  async extractStructuredData(content) {
    try {
      const response = await this.openaiClient.chat.completions.create({
        model: "gpt-3.5-turbo",
        messages: [{
          role: "system",
          content: "Extract structured travel information from content. Return JSON with title, location, activities, tips, and summary."
        }, {
          role: "user",
          content: content.substring(0, 2000)
        }],
        max_tokens: 500
      });

      return JSON.parse(response.choices[0].message.content);
    } catch (error) {
      console.error('Structured data extraction failed:', error);
      return {
        title: this.extractTitle(content),
        location: this.extractLocation(content),
        summary: content.substring(0, 200) + '...'
      };
    }
  }
}

            }
          }
        ],
        function_call: { name: "analyze_travel_content" }
      });

      const analysis = JSON.parse(response.choices[0].message.function_call.arguments);
      return {
        ...analysis,
        confidence: this.calculateConfidence(response),
        processedAt: new Date().toISOString()
      };
    } catch (error) {
      console.error('Content analysis failed:', error);
      throw new Error('Failed to analyze content with AI');
    }
  }

  async generateRecommendations(userPreferences, travelHistory) {
    const prompt = this.buildRecommendationPrompt(userPreferences, travelHistory);
    
    try {
      const response = await this.openai.chat.completions.create({
        model: "gpt-4",
        messages: [
          {
            role: "system",
            content: "You are a personalized travel recommendation engine. Generate tailored travel suggestions based on user preferences and history."
          },
          {
            role: "user",
            content: prompt
          }
        ],
        max_tokens: 800,
        temperature: 0.7
      });

      return this.parseRecommendations(response.choices[0].message.content);
    } catch (error) {
      console.error('Recommendation generation failed:', error);
      throw new Error('Failed to generate AI recommendations');
    }
  }

  calculateConfidence(response) {
    // Calculate confidence based on response characteristics
    const usage = response.usage;
    const completionRatio = usage.completion_tokens / usage.total_tokens;
    return Math.min(completionRatio * 2, 1); // Normalize to 0-1
  }

// hooks/useAPI.ts
import { useState, useEffect } from 'react';
import { useTravelStore } from '../store/travelStore';

interface APIOptions {
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  body?: any;
  headers?: Record;
  cache?: boolean;
  offline?: boolean;
}

export function useAPI(endpoint: string, options: APIOptions = {}) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const { isOnline } = useTravelStore();

  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        setError(null);

        // Check offline mode
        if (!isOnline && options.offline) {
          const cachedData = await getCachedData(endpoint);
          if (cachedData) {
            setData(cachedData);
            return;
          }
        }

        const response = await fetch(`${process.env.REACT_APP_API_URL}${endpoint}`, {
          method: options.method || 'GET',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${getAuthToken()}`,
            ...options.headers
          },
          body: options.body ? JSON.stringify(options.body) : undefined
        });

        if (!response.ok) {
          throw new Error(`API Error: ${response.status} ${response.statusText}`);
        }

        const result = await response.json();
        setData(result.data);

        // Cache successful responses
        if (options.cache && response.ok) {
          await cacheData(endpoint, result.data);
        }

      } catch (err) {
        const errorMessage = err instanceof Error ? err.message : 'Unknown error';
        setError(errorMessage);
        
        // Try to load cached data on error
        if (options.offline) {
          const cachedData = await getCachedData(endpoint);
          if (cachedData) {
            setData(cachedData);
            setError(`Using cached data: ${errorMessage}`);
          }
        }
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, [endpoint, options.method, JSON.stringify(options.body)]);

  return { data, loading, error, refetch: () => fetchData() };
}

// Utility functions
async function getCachedData(key: string): Promise {
  try {
    const cached = localStorage.getItem(`api_cache_${key}`);
    if (cached) {
      const { data, timestamp } = JSON.parse(cached);
      const isExpired = Date.now() - timestamp > 5 * 60 * 1000; // 5 minutes
      if (!isExpired) {
        return data;
      }
    }
  } catch (error) {
    console.error('Cache retrieval error:', error);
  }
  return null;
}

async function cacheData(key: string, data: T): Promise {
  try {
    const cacheEntry = {
      data,
      timestamp: Date.now()
    };
    localStorage.setItem(`api_cache_${key}`, JSON.stringify(cacheEntry));
  } catch (error) {
    console.error('Cache storage error:', error);
  }
}

function getAuthToken(): string {
  return localStorage.getItem('auth_token') || '';
}

// middleware/errorHandler.js
export const errorHandler = (error, request, reply) => {
  const { statusCode = 500, message } = error;
  
  // Log error details
  request.log.error({
    error: error.message,
    stack: error.stack,
    url: request.url,
    method: request.method,
    headers: request.headers,
    body: request.body
  });

  // Determine error response based on environment
  const errorResponse = {
    success: false,
    error: {
      message: statusCode < 500 ? message : 'Internal server error',
      code: error.code || 'UNKNOWN_ERROR',
      timestamp: new Date().toISOString(),
      requestId: request.id
    }
  };

  // Add stack trace in development
  if (process.env.NODE_ENV === 'development') {
    errorResponse.error.stack = error.stack;
  }

  reply.code(statusCode).send(errorResponse);
};

// utils/OfflineImageManager.ts
class OfflineImageManager {
  private cache: Cache | null = null;
  private db: IDBDatabase | null = null;

  async initialize() {
    // Initialize Cache API
    if ('caches' in window) {
      this.cache = await caches.open('travel-images-v1');
    }

    // Initialize IndexedDB for metadata
    return new Promise((resolve, reject) => {
      const request = indexedDB.open('TravelImagesDB', 1);
      
      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve();
      };
      
      request.onupgradeneeded = (event) => {
        const db = (event.target as IDBOpenDBRequest).result;
        const store = db.createObjectStore('images', { keyPath: 'url' });
        store.createIndex('lastAccessed', 'lastAccessed');
        store.createIndex('size', 'size');
      };
    });
  }

  async cacheImage(url: string, priority: 'high' | 'medium' | 'low' = 'medium') {
    if (!this.cache) return false;

    try {
      // Check if already cached
      const cachedResponse = await this.cache.match(url);
      if (cachedResponse) return true;

      // Fetch and cache the image
      const response = await fetch(url);
      if (response.ok) {
        await this.cache.put(url, response.clone());
        
        // Store metadata in IndexedDB
        await this.storeImageMetadata(url, {
          size: parseInt(response.headers.get('content-length') || '0'),
          type: response.headers.get('content-type') || 'image/jpeg',
          cachedAt: Date.now(),
          lastAccessed: Date.now(),
          priority
        });
        
        return true;
      }
    } catch (error) {
      console.error('Failed to cache image:', url, error);
    }
    
    return false;
  }

  async getImage(url: string): Promise {
    if (!this.cache) return null;

    try {
      const cachedResponse = await this.cache.match(url);
      if (cachedResponse) {
        // Update last accessed time
        await this.updateLastAccessed(url);
        
        const blob = await cachedResponse.blob();
        return URL.createObjectURL(blob);
      }
    } catch (error) {
      console.error('Failed to retrieve cached image:', url, error);
    }
    
    return null;
  }

  async preloadImages(urls: string[], maxConcurrent = 3) {
    const chunks = this.chunkArray(urls, maxConcurrent);
    
    for (const chunk of chunks) {
      await Promise.allSettled(
        chunk.map(url => this.cacheImage(url, 'low'))
      );
    }
  }

  async cleanupCache(maxSizeMB = 100) {
    if (!this.db || !this.cache) return;

    const transaction = this.db.transaction(['images'], 'readonly');
    const store = transaction.objectStore('images');
    const index = store.index('lastAccessed');
    
    const images: any[] = [];
    let totalSize = 0;

    return new Promise((resolve) => {
      index.openCursor().onsuccess = async (event) => {
        const cursor = (event.target as IDBRequest).result;
        
        if (cursor) {
          images.push(cursor.value);
          totalSize += cursor.value.size;
          cursor.continue();
        } else {
          // Remove oldest images if over limit
          const maxSize = maxSizeMB * 1024 * 1024;
          if (totalSize > maxSize) {
            images.sort((a, b) => a.lastAccessed - b.lastAccessed);
            
            let removedSize = 0;
            for (const image of images) {
              if (totalSize - removedSize <= maxSize) break;
              
              await this.cache!.delete(image.url);
              await this.removeImageMetadata(image.url);
              removedSize += image.size;
            }
          }
          resolve();
        }
      };
    });
  }

  private chunkArray(array: T[], size: number): T[][] {
    const chunks: T[][] = [];
    for (let i = 0; i < array.length; i += size) {
      chunks.push(array.slice(i, i + size));
    }
    return chunks;
  }

  private async storeImageMetadata(url: string, metadata: any) {
    if (!this.db) return;
    
    const transaction = this.db.transaction(['images'], 'readwrite');
    const store = transaction.objectStore('images');
    await store.put({ url, ...metadata });
  }

  private async updateLastAccessed(url: string) {
    if (!this.db) return;
    
    const transaction = this.db.transaction(['images'], 'readwrite');
    const store = transaction.objectStore('images');
    const image = await store.get(url);
    
    if (image) {
      image.lastAccessed = Date.now();
      await store.put(image);
    }
  }

  private async removeImageMetadata(url: string) {
    if (!this.db) return;
    
    const transaction = this.db.transaction(['images'], 'readwrite');
    const store = transaction.objectStore('images');
    await store.delete(url);
  }
}
